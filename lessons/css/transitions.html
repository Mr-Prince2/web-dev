üß™ Example: Slide Up Caption

.image-box {
  position: relative;
  overflow: hidden;
}
.caption {
  position: absolute;
  bottom: -50px;
  background: rgba(0, 0, 0, 0.7);
  color: white;
  padding: 10px;
  transition: bottom 0.4s ease;
}
.image-box:hover .caption {
  bottom: 0;
}
